#### 1. 求最大公约数和最小公倍数

```c
#include <stdio.h>
int GreatestCommonDivisor(int m, int n) {
    int tmp;
    while (tmp = m % n) {
        m = n;
        n = tmp;
    }
    return n;
}
int LeastCommonMultiple(int m, int n) {
    return m * n / GreatestCommonDivisor(m, n);	// 函数的嵌套调用, 但是函数不能嵌套定义
}
int main() {
    int a, b;
    puts("请输入两个数:");
    scanf("%d%d", &a, &b);
    printf("最大公约数是: %d\n", GreatestCommonDivisor(a, b));
    printf("最小公倍数数是: %d\n", LeastCommonMultiple(a, b));
    return 0;
}
```

#### 2.求ax²+bx+c=0的根

```c
#include <math.h>
#include <stdio.h>
#include <stdlib.h>
void UnaryQuadraticEquationRoot(double a, double b, double c) {
    if (fabs(a) <= 1e-15) {
        puts("不是一元二次方程");
        exit(0); // 不是一元二次方程就退出
    } else {
        double deerta = b * b - 4 * a * c; // 计算Δ
        double p = -b / (2 * a);
        double q = sqrt(deerta) / (2 * a);
        if (fabs(deerta) <= 1e-15)
            printf("该方程有两个相等的根为:\nx = %.4lf\n", p);
        else if (deerta > 1e-15) {
            double x1 = p + q;
            double x2 = p - q;
            printf("该方程有两个不等的实根为:\nx1 = %.4lf , x 2= %.4lf\n", x1, x2);
        } else {
            double realpart = -b / (2 * a);            // 实部
            double imagpart = sqrt(-deerta) / (2 * a); // 虚部
            printf("该方程有两个共轭的复根:\n");
            printf("x1 = %.4lf+ %5.4lfi\n", realpart, imagpart);
            printf("x2 = %.4lf- %5.4lfi", realpart, imagpart);
        }
    }
}
int main() {
    double a, b, c;
    puts("请输入一元二次方程的系数:");
    scanf("%lf %lf %lf", &a, &b, &c);
    UnaryQuadraticEquationRoot(a, b, c);
    return 0;
}
```
